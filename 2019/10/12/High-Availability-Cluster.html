Add New NameNode to HA HDFS

HDFS 2.7 Expected exactly 2 NameNodes in namespace.

vi /usr/hdp/current/hadoop-client/etc/hadoop/hdfs-site.xml
hawq scp -f ~/hostfile /usr/hdp/current/hadoop-client/etc/hadoop/hdfs-site.xml =:/tmp/
hawq ssh -f ~/hostfile 'sudo cp /tmp/hdfs-site.xml /usr/hdp/current/hadoop-client/etc/hadoop/hdfs-site.xml'


hawq ssh -h STANDBYHOST -e 'sudo -u hdfs -E /usr/hdp/current/hadoop-client/sbin/hadoop-daemon.sh stop namenode'


hawq ssh -h STANDBYHOST -e 'sudo -u hdfs -E /usr/hdp/current/hadoop-client/sbin/hadoop-daemon.sh start journalnode'

hawq ssh -h STANDBYHOST -e 'sudo -u hdfs -E hdfs namenode -bootstrapStandby'
hawq ssh -h STANDBYHOST -e 'sudo -u hdfs -E /usr/hdp/current/hadoop-client/sbin/hadoop-daemon.sh start namenode'

hawq ssh -h STANDBYHOST -e 'sudo -u zookeeper /usr/hdp/current/zookeeper-server/bin/zkServer.sh start'
hawq ssh -h STANDBYHOST -e 'sudo -u hdfs -E /usr/hdp/current/hadoop-client/sbin/hadoop-daemon.sh start zkfc'

hawq ssh -h STANDBYHOST -e 'sudo -u hdfs -E /usr/hdp/current/hadoop-client/sbin/hadoop-daemon.sh start journalnode'
hawq ssh -h STANDBYHOST -e 'sudo -u hdfs -E /usr/hdp/current/hadoop-client/sbin/hadoop-daemon.sh start zkfc'
