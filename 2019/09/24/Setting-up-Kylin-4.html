Add Local Apt Repo

sudo mount /dev/sr0 /media/kylin/Kylin-4.0.2
cp -r /media/kylin/Kylin-4.0.2 /tmp
sudo chown -R _apt:root /tmp/Kylin-4.0.2
echo 'deb file:///tmp/Kylin-4.0.2 juniper main' | sudo tee /etc/apt/sources.list.d/Kylin-4.0.2.list
sudo apt-get update

Toolchain

# There is no gcc-c++ ...
cd ~/dev/
export PREFIX=/home/kylin/dev/package
export MAKEFLAGS=-j$((`nproc`+3))

sudo apt-get install gcc-4.9 libstdc++-4.9-dev git libncurses5-dev gawk libreadline-dev

# Clang
# The newer version depends on GLIBCXX_3.4.22.
wget https://github.com/llvm/llvm-project/releases/download/llvmorg-7.1.0/clang+llvm-7.1.0-aarch64-linux-gnu.tar.xz

# CMake
wget https://github.com/Kitware/CMake/releases/download/v3.15.3/cmake-3.15.3.tar.gz
tar xf cmake-3.15.3.tar.gz
cd cmake-3.15.3
./bootstrap -- -DCMAKE_BUILD_TYPE:STRING=Release
make

# Thirdparty
make check-target-dir
make build-hawq-dep

wget -O config.guess 'http://git.savannah.gnu.org/gitweb/?p=config.git;a=blob_plain;f=config.guess;hb=HEAD'
wget -O config.sub 'http://git.savannah.gnu.org/gitweb/?p=config.git;a=blob_plain;f=config.sub;hb=HEAD'
find src/ -name config.sub -exec cp config.sub {} \;
find src/ -name config.guess -exec cp config.guess {} \;

# Apache HAWQ
source $PREFIX/env.sh
wget -O config/config.guess 'http://git.savannah.gnu.org/gitweb/?p=config.git;a=blob_plain;f=config.guess;hb=HEAD'
git grep -l 'if (HWCrc32c::available())'  | xargs sed -i 's/if (HWCrc32c::available())/if(false)/'
git grep -l 'if (dbcommon::HWCrc32c::available())'  | xargs sed -i 's/if (dbcommon::HWCrc32c::available())/if(false)/'
sudo install -o $USER -d /usr/local/hawq
export DEPENDENCY_PATH=$PREFIX
./configure --with-python --disable-spinlocks
make DEFAULTTARGET= install



# PostgreSQL
sudo install -o $USER -d /usr/local/postgres
./configure --disable-spinlocks --prefix=/usr/local/postgres
make install
export PATH=/usr/local/postgres/bin:$PATH

git grep -l 'if (HWCrc32c::available())'  | xargs sed -i 's/if (HWCrc32c::available())/if(false)/'

export RUN_UNITTEST=no
